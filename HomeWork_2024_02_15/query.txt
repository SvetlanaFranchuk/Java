CREATE TABLE students(
                            id BIGINT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
                            firstName VARCHAR(100) NOT NULL CHECK (firstName <> ''),
                            lastName VARCHAR(100) NOT NULL CHECK (lastName <> ''),
                            studyClass BIGINT, FOREIGN KEY (studyClass) REFERENCES study_classes("id")
);

                        subject BIGINT, FOREIGN KEY (subject) REFERENCES subjects(id),
                        grade INT NOT NULL CHECK (grade > 0 AND  grade <=12)
